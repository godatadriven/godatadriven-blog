def generate_row(uid, high_card, low_card):
    c = CENTROIDS[int(rnd.rand() * 4)]
    return (
        uid,
        int(rnd.rand() * high_card),
        int(rnd.rand() * low_card),
        int(rnd.pareto(6.0) * high_card),
        int(rnd.randn() * high_card),
        rnd.randn() + c[0],
        rnd.randn() + c[1]
    )

def generate_rows(start, n):
    for x in xrange(start, start + n):
        yield generate_row(start * n + (x - start), HIGH, LOW)

parsers = (int, int, int, int, int, float, float)
# Read back the file, because using the original RDD directly might partially re-generate random numbers
data = sc\
.textFile('/mnt/raid/numbers.txt/part*')\
.map(lambda line: map(unicode.strip, line.split('|')))\
.map(lambda d: [parser(d) for parser, d in zip(parsers, d)])

df = pd.DataFrame(
    data.sample(False, float(NUM_SAMPLES) / NUM_ROWS).collect(),
