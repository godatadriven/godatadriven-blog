returns = cypher(graph, '
match (g:Group)-[:ORGANISED]->(e:Event)<-[r:RSVP_ED]-(m:Member)
where r.response = "yes"
with g, count(distinct m) as total
match g-[:ORGANISED]->(e:Event)<-[r:RSVP_ED]-(m:Member)
where r.response="yes"
with g as group, total, m, count(*) > 1 as return_visitor
return
    group.name as group,
    group.created as created,
    total,
    return_visitor,
    count(*) as cnt
order by group, return_visitor
')
